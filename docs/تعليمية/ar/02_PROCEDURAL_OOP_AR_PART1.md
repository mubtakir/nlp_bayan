## المتغيرات وأنواع البيانات

```bayan
هجين {
    # أرقام
    عدد = 42
    ط = 3.14159

    # نصوص
    اسم = "أحمد"
    مدينة = "الرياض"

    # قوائم
    عناصر = [1, 2, 3]

    # قواميس
    شخص = {
        "الاسم": "علي",
        "العمر": 25
    }
}
```

## العمليات الحسابية والمنطقية

```bayan
هجين {
    أ = 10
    ب = 3

    مجموع = أ + ب
    فرق = أ - ب
    حاصل_ضرب = أ * ب
    قسمة = أ / ب

    شرط١ = (مجموع > 10) و (ب < 5)
    شرط٢ = ليس (أ == ب)
}
```

## نصوص وقوائم وقواميس (مختصر)

```bayan
هجين {
    نص = "مرحبا بيان"
    اطبع(upper(نص))

    قائمة = ["أ", "ب", "ج"]
    اطبع(len(قائمة))

    مستخدم = {
        "الاسم": "سارة",
        "النتيجة": 0.9
    }
    اطبع(مستخدم["الاسم"] + ": " + str(مستخدم["النتيجة"]))
}
```
