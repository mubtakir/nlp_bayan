## 7. الشروط (if/elif/else)

### 7.1 الشرط البسيط

```bayan
هجين {
    العمر = 18

    اذا العمر >= 18: {
        اطبع("بالغ")
    }
}
```

### 7.2 if-else

```bayan
هجين {
    الحرارة = 30

    اذا الحرارة > 25: {
        اطبع("الجو حار")
    }
    والا: {
        اطبع("الجو معتدل")
    }
}
```

### 7.3 if-elif-else

```bayan
هجين {
    الدرجة = 85

    اذا الدرجة >= 90: {
        اطبع("ممتاز")
    }
    والا_اذا الدرجة >= 80: {
        اطبع("جيد جداً")
    }
    والا_اذا الدرجة >= 70: {
        اطبع("جيد")
    }
    والا_اذا الدرجة >= 60: {
        اطبع("مقبول")
    }
    والا: {
        اطبع("راسب")
    }
}
```

### 7.4 شروط متداخلة

```bayan
هجين {
    العمر = 20
    لديه_رخصة = صحيح

    اذا العمر >= 18: {
        اذا لديه_رخصة: {
            اطبع("يمكنك القيادة")
        }
        والا: {
            اطبع("تحتاج رخصة قيادة")
        }
    }
    والا: {
        اطبع("أنت صغير السن")
    }
}
```

## 8. الحلقات (for/while)

### 8.1 حلقة for

```bayan
هجين {
    # التكرار على قائمة
    فواكه = ["تفاح", "موز", "برتقال"]

    لكل فاكهة في فواكه: {
        اطبع(فاكهة)
    }

    # التكرار على range
    لكل ع في range(5): {
        اطبع(ع)  # 0, 1, 2, 3, 4
    }

    # range مع بداية ونهاية
    لكل ع في range(1, 6): {
        اطبع(ع)  # 1, 2, 3, 4, 5
    }

    # range مع خطوة
    لكل ع في range(0, 10, 2): {
        اطبع(ع)  # 0, 2, 4, 6, 8
    }
}
```

### 8.2 حلقة while

```bayan
هجين {
    # حلقة while بسيطة
    العداد = 0

    بينما العداد < 5: {
        اطبع(العداد)
        العداد = العداد + 1
    }

    # حلقة while مع شرط
    رقم = 1

    بينما رقم <= 10: {
        اذا رقم % 2 == 0: {
            اطبع(رقم)
        }
        رقم = رقم + 1
    }
}
```

### 8.3 break و continue

```bayan
هجين {
    # break - إيقاف الحلقة
    لكل ع في range(10): {
        اذا ع == 5: {
            اكسر
        }
        اطبع(ع)  # 0, 1, 2, 3, 4
    }

    # continue - تخطي التكرار الحالي
    لكل ع في range(5): {
        اذا ع == 2: {
            استمر
        }
        اطبع(ع)  # 0, 1, 3, 4
    }
}
```

### 8.4 enumerate و zip

```bayan
هجين {
    # enumerate - الحصول على الفهرس والقيمة
    فواكه = ["تفاح", "موز", "برتقال"]

    لكل فهرس_قيمة في enumerate(فواكه): {
        فهرس = فهرس_قيمة[0]
        قيمة = فهرس_قيمة[1]
        اطبع(فهرس)
        اطبع(قيمة)
    }

    # zip - دمج قائمتين
    أسماء = ["أحمد", "فاطمة", "علي"]
    أعمار = [25, 22, 30]

    لكل زوج في zip(أسماء, أعمار): {
        اسم = زوج[0]
        عمر = زوج[1]
        اطبع(اسم)
        اطبع(عمر)
    }
}
```

## 9. الدوال (Functions)

### 9.1 تعريف دالة بسيطة

```bayan
هجين {
    دالة تحية(): {
        اطبع("مرحباً!")
    }

    تحية()  # استدعاء الدالة
}
```

### 9.2 دالة بمعاملات

```bayan
هجين {
    دالة تحية_شخص(اسم): {
        اطبع("مرحباً " + اسم)
    }

    تحية_شخص("أحمد")  # "مرحباً أحمد"
    تحية_شخص("فاطمة")  # "مرحباً فاطمة"
}
```

### 9.3 دالة بقيمة إرجاع

```bayan
هجين {
    دالة جمع(أ, ب): {
        ارجع أ + ب
    }

    النتيجة = جمع(10, 20)
    اطبع(النتيجة)  # 30
}
```

### 9.4 معاملات افتراضية

```bayan
هجين {
    دالة تحية(اسم, تحية="مرحباً"): {
        ارجع تحية + " " + اسم
    }

    رسالة1 = تحية("أحمد")
    اطبع(رسالة1)  # "مرحباً أحمد"

    رسالة2 = تحية("فاطمة", "أهلاً")
    اطبع(رسالة2)  # "أهلاً فاطمة"
}
```

### 9.5 دوال متعددة القيم

```bayan
هجين {
    def get_min_max(numbers): {
        minimum = min(numbers)
        maximum = max(numbers)
        return [minimum, maximum]
    }

    result = get_min_max([5, 2, 8, 1, 9])
    print(result[0])  # 1 (minimum)
    print(result[1])  # 9 (maximum)
}
```

### 9.6 دوال متداخلة

```bayan
هجين {
    def outer_function(x): {
        def inner_function(y): {
            return x + y
        }

        return inner_function(10)
    }

    result = outer_function(5)
    print(result)  # 15
}
```

## 10. معالجة الاستثناءات

### 10.1 try-except الأساسي

```bayan
هجين {
    try: {
        x = 10 / 0
    }
    except ZeroDivisionError: {
        print("خطأ: القسمة على صفر!")
    }
}
```

### 10.2 معالجة استثناءات متعددة

```bayan
هجين {
    try: {
        number = int("abc")
    }
    except ValueError: {
        print("خطأ: قيمة غير صحيحة")
    }
    except TypeError: {
        print("خطأ: نوع غير صحيح")
    }
}
```

### 10.3 try-except-finally

```bayan
هجين {
    try: {
        file = open("data.txt", "r")
        content = file.read()
    }
    except FileNotFoundError: {
        print("الملف غير موجود")
    }
    finally: {
        print("تم الانتهاء من المحاولة")
    }
}
```

### 10.4 رفع استثناء

```bayan
هجين {
    def check_age(age): {
        if (age < 0) {
            raise ValueError("العمر لا يمكن أن يكون سالباً")
        }
        return age
    }

    try: {
        check_age(-5)
    }
    except ValueError: {
        print("خطأ في العمر")
    }
}
```
