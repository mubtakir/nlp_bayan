## 11. الأصناف والكائنات

### 11.1 تعريف صنف بسيط

```bayan
هجين {
    صنف شخص: {
        دالة __init__(الذات, اسم, عمر): {
            الذات.اسم = اسم
            الذات.عمر = عمر
        }

        دالة عرض(الذات): {
            اطبع(الذات.اسم)
            اطبع(الذات.عمر)
        }
    }

    # إنشاء كائن
    شخص1 = شخص("أحمد", 25)
    شخص1.عرض()
}
```

### 11.2 الخصائص والطرق

```bayan
هجين {
    صنف طالب: {
        دالة __init__(الذات, اسم, درجة): {
            الذات.اسم = اسم
            الذات.درجة = درجة
            الذات.مواد = []
        }

        دالة اضف_مادة(الذات, مادة): {
            الذات.مواد.append(مادة)
        }

        دالة احصل_على_المعلومات(الذات): {
            معلومات = الذات.اسم + " - الدرجة: " + str(الذات.درجة)
            ارجع معلومات
        }
    }

    طالب = طالب("فاطمة", 95)
    طالب.اضف_مادة("رياضيات")
    طالب.اضف_مادة("فيزياء")

    اطبع(طالب.احصل_على_المعلومات())
    اطبع(طالب.مواد)
}
```


### 11.3 صنف مع عدة طرق

```bayan
هجين {
    صنف حساب_بنكي: {
        دالة __init__(الذات, المالك, الرصيد): {
            الذات.المالك = المالك
            الذات.الرصيد = الرصيد
        }

        دالة ايداع(الذات, المبلغ): {
            الذات.الرصيد = الذات.الرصيد + المبلغ
            ارجع الذات.الرصيد
        }

        دالة سحب(الذات, المبلغ): {
            اذا المبلغ > الذات.الرصيد: {
                اطبع("رصيد غير كافٍ")
                ارجع الذات.الرصيد
            }
            الذات.الرصيد = الذات.الرصيد - المبلغ
            ارجع الذات.الرصيد
        }

        دالة احصل_على_الرصيد(الذات): {
            ارجع الذات.الرصيد
        }
    }

    حساب = حساب_بنكي("أحمد", 1000)
    حساب.ايداع(500)
    اطبع(حساب.احصل_على_الرصيد())  # 1500

    حساب.سحب(200)
    اطبع(حساب.احصل_على_الرصيد())  # 1300
}
```

## 12. الوراثة

### 12.1 وراثة بسيطة

```bayan
هجين {
    # الصنف الأب
    صنف حيوان: {
        دالة __init__(الذات, اسم): {
            الذات.اسم = اسم
        }

        دالة تكلم(الذات): {
            اطبع("الحيوان يصدر صوتاً")
        }
    }

    # الصنف الابن
    صنف كلب: {
        دالة __init__(الذات, اسم, سلالة): {
            الذات.اسم = اسم
            الذات.سلالة = سلالة
        }

        دالة تكلم(الذات): {
            اطبع("الكلب ينبح")
        }
    }

    كلب = كلب("ريكس", "جيرمن")
    كلب.تكلم()  # "الكلب ينبح"
}
```

### 12.2 استخدام super()

```bayan
هجين {
    صنف مركبة: {
        دالة __init__(الذات, العلامة, الطراز): {
            الذات.العلامة = العلامة
            الذات.الطراز = الطراز
        }

        دالة معلومات(الذات): {
            ارجع الذات.العلامة + " " + الذات.الطراز
        }
    }

    صنف سيارة: {
        دالة __init__(الذات, العلامة, الطراز, الأبواب): {
            الذات.العلامة = العلامة
            الذات.الطراز = الطراز
            الذات.الأبواب = الأبواب
        }

        دالة معلومات(الذات): {
            معلومات_أساسية = الذات.العلامة + " " + الذات.الطراز
            ارجع معلومات_أساسية + " - الأبواب: " + str(الذات.الأبواب)
        }
    }

    سيارة = سيارة("تويوتا", "كامري", 4)
    اطبع(سيارة.معلومات())
}
```

## 13. التغليف

### 13.1 خصائص خاصة (Private)

```bayan
هجين {
    صنف حساب_بنكي: {
        دالة __init__(الذات, رصيد): {
            الذات.__رصيد = رصيد  # خاصية خاصة
        }

        دالة احصل_على_الرصيد(الذات): {
            ارجع الذات.__رصيد
        }

        دالة عين_الرصيد(الذات, مبلغ): {
            اذا مبلغ >= 0: {
                الذات.__رصيد = مبلغ
            }
        }
    }

    حساب = حساب_بنكي(1000)
    اطبع(حساب.احصل_على_الرصيد())  # 1000

    حساب.عين_الرصيد(2000)
    اطبع(حساب.احصل_على_الرصيد())  # 2000
}
```

## 14. تعدد الأشكال

### 14.1 Polymorphism الأساسي

```bayan
هجين {
    صنف شكل: {
        دالة مساحة(الذات): {
            ارجع 0
        }
    }

    صنف مستطيل: {
        دالة __init__(الذات, العرض, الارتفاع): {
            الذات.العرض = العرض
            الذات.الارتفاع = الارتفاع
        }

        دالة مساحة(الذات): {
            ارجع الذات.العرض * الذات.الارتفاع
        }
    }

    صنف دائرة: {
        دالة __init__(الذات, نصف_القطر): {
            الذات.نصف_القطر = نصف_القطر
        }

        دالة مساحة(الذات): {
            ارجع 3.14 * الذات.نصف_القطر * الذات.نصف_القطر
        }
    }

    مستطيل = مستطيل(5, 10)
    اطبع(مستطيل.مساحة())  # 50

    دائرة = دائرة(7)
    اطبع(دائرة.مساحة())  # 153.86
}
```
