# تشابه نصّي باستخدام TFIDF ناعم

```bayan
include "ai/nlp.bayan"

نصوص = [
  "القط يجلس على السجادة",
  "الكلب يجلس على الخشبة",
  "القطط والكلاب أصدقاء"
]

نموذج = بناء_TFIDF_ناعم(نصوص, عتبة=0.9)
تشابه = تشابه_جيبي_TFIDF_ناعم("القط على السجادة", "الكلب على الخشبة", نموذج)

print("التشابه:")
print(تشابه)
```
